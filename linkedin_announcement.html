quick_start.py Switch Provider in 2 Lines 
from ai_bridge_kit import AIClient
from ai_bridge_kit.providers import OpenAIProvider, AnthropicProvider, OllamaProvider

# ── One client. Any provider. Zero rewrites. ──────────────────────────────────

client = AIClient()

# OpenAI
client.register_provider(OpenAIProvider(api_key="sk-..."), set_default=True)

# Swap to Anthropic  << only this line changes
client.register_provider(AnthropicProvider(api_key="sk-ant-..."), set_default=True)

# Or go fully local with Ollama  << no API keys, no cost
client.register_provider(OllamaProvider(base_url="http://localhost:11434"), set_default=True)

# Your business logic stays IDENTICAL regardless of provider
response = client.chat("Summarize agentic AI in one sentence.")
print(response.content)

# Embeddings work the same way
emb = client.embed(["ai-bridge-kit", "unified api"])
print(len(emb.vectors[0]), "dimensions")
